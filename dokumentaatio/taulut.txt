CREATE TABLE [Node] (
	[NodeId] INTEGER NOT NULL PRIMARY KEY,
	[Latitude] REAL NOT NULL,
	[Longitude] REAL NOT NULL);
CREATE TABLE [Kaari] (
	[KaariId] INTEGER  PRIMARY KEY  NOT NULL,
	[NodeId1] INTEGER  NOT NULL,
	[NodeId2] INTEGER  NOT NULL,
	[Pituus] REAL NOT NULL,
	FOREIGN KEY (NodeId1) REFERENCES Node(NodeId),
	FOREIGN KEY (NodeId2) REFERENCES Node(NodeId));
